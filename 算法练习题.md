# 算法练习

## 要求不用数学库，求 sqrt (2)精确到小数点后10位(Java版)](https://www.cnblogs.com/zhazhaacmer/p/11052672.html)

### 核心思路

- 二分查找法

### 代码

```java
public static double getSqrt(int num) {
        double low = 1;
        double high = 2;
        double mid = 1;
        while (high - low > 1 * Math.pow(10, -10)) {
            mid = (low + high) * 0.5;
            if (mid * mid <= num) {
                low = mid;
            } else {
                high = mid;
            }
        }
        return mid;
    }
```

## 快速排序

### 单路快排

```java
public void sort(int[] arr) {
        sort(0, arr.length - 1, arr);
    }

    public void sort(int low, int high, int[] arr) {
        if (low >= high) {
            return;
        }
        int p = partition(low, high, arr);
        sort(low, p - 1, arr);
        sort(p + 1, high, arr);
    }

    private int partition(int low, int high, int[] arr) {
        int j = low;
        for (int i = low + 1; i <= high; i++) {
            if (arr[i] < arr[low]) {
                j++;
                swap(i, j, arr);
            }
        }
        swap(j, low, arr);
        return j;
    }

    private void swap(int i, int j, int[] arr) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
```

```java
    private int partition(int low, int high, int[] arr) {
        int i = low + 1, j = high;
        while (true) {
            while (i <= high && arr[i] < arr[low]) {
                i++;
            }
            while (j >= low + 1 && arr[j] > arr[low]) {
                j--;
            }
            if (i > j)
                break;
            swap(i, j, arr);
            i++;
            j--;
        }
        swap(j, low, arr);
        return j;
    }

    private void swap(int i, int j, int[] arr) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
```

### 三路快排

```java
private void sort(int[] arr) {
        sort(0, arr.length - 1, arr);
    }

    private void sort(int low, int high, int[] arr) {
        if (low >= high) {
            return;
        }
        //lt是从头到尾第一个小于pilot的值；gt是从尾到头第一个大于pilot的值
        int lt = low, gt = high + 1;
        int i = low + 1;

        while (i < gt) {
            if (arr[i] < arr[low]) {
                swap(i, lt + 1, arr);
                i++;
                lt++;
            } else if (arr[i] > arr[low]) {
                swap(i, gt - 1, arr);
                gt--;
            } else {
                i++;
            }
        }
        swap(low, lt, arr);
        sort(low, lt - 1, arr);
        sort(gt, high, arr);
    }

    private void swap(int i, int j, int[] arr) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
```

### 快排获取第n大的元素

```java
private static int quickSortNum(int[] arr, int k) {
        return quickSortNum(arr, 0, arr.length - 1, k - 1);
    }

    private static int quickSortNum(int[] arr, int low, int high, int k) {
        int p = partition(low, high, arr);
        if (p > k) {
            return quickSortNum(arr, low, p - 1, k);
        } else if (p < k) {
            return quickSortNum(arr, p + 1, high, k);
        }
        return arr[p];
    }

    private static int partition(int low, int high, int[] arr) {
        int i = low + 1, j = high;
        while (true) {
            while (i <= high && arr[i] < arr[low]) {
                i++;
            }
            while (j >= low + 1 && arr[j] > arr[low]) {
                j--;
            }
            if (i > j)
                break;
            swap(i, j, arr);
            i++;
            j--;
        }
        swap(j, low, arr);
        return j;
    }

    private static void swap(int i, int j, int[] arr) {
        int tmp = arr[i];
        arr[i] = arr[j];
        arr[j] = tmp;
    }
```
